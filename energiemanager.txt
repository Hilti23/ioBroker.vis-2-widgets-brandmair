// Generische Getter mit Fallback auf Device-Konfiguration
function g(deviceId, prop, fallback) {
    try {
        const val = getState(`${DP_BASE}.${deviceId}.${prop}`).val;
        if (val === null || val === undefined) return fallback;
        return val;
    } catch (e) { return fallback; }
}

function gJson(deviceId, prop, fallback) {
    try {
        const val = getState(`${DP_BASE}.${deviceId}.${prop}`).val;
        if (!val || val === '[]' || val === '' || val === 'null') return fallback;
        return JSON.parse(val);
    } catch (e) { return fallback; }
}

function getSurplus() { return getState(CONFIG.pv_grid_dp).val * -1; }
function getSOC() { return getState(CONFIG.battery_soc_dp).val; }
function getPVPower() { return (getState(CONFIG.pv_dc_string1_dp).val || 0) + (getState(CONFIG.pv_dc_string2_dp).val || 0); }

// RCT liefert negativ bei Laden, daher * -1
function getBatteryChargePower() {
    try { return (getState(CONFIG.battery_power_dp).val || 0) * -1; }
    catch (e) { return 0; }
}

function getForecastTodayKwh() {
    let fc = 0, sc = 0, cnt = 0;
    try { fc = getState(CONFIG.forecast_solar_today_dp).val || 0; if (fc > 0) cnt++; } catch (e) { }
    try { sc = getState(CONFIG.forecast_solcast_today_dp).val || 0; if (sc > 0) cnt++; } catch (e) { }
    return cnt === 0 ? 0 : (fc + sc) / cnt / 1000;
}

function getAvgSurplus(minutes) {
    const cutoff = Date.now() - (minutes * 60000);
    const rel = state.surplus_history.filter(v => v.ts >= cutoff);
    return rel.length === 0 ? 0 : rel.reduce((a, b) => a + b.value, 0) / rel.length;
}

function getWeekday() { const d = new Date().getDay(); return d === 0 ? 7 : d; }

function isInTimeWindow(s, e) {
    if (!s || !e) return false;
    const [sH, sM] = s.split(':').map(Number);
    const [eH, eM] = e.split(':').map(Number);
    const now = new Date().getHours() * 60 + new Date().getMinutes();
    return now >= (sH * 60 + sM) && now < (eH * 60 + eM);
}

function isInSchedule(d) {
    const wd = getWeekday();
    // Try time_rules first
    const rules = gJson(d.id, 'time_rules', null);
    if (rules && Array.isArray(rules) && rules.length > 0) {
        for (const r of rules) {
            if (r.days && r.days.includes(wd)) {
                return isInTimeWindow(r.start, r.end);
            }
        }
        // Current weekday not in any rule = inactive
        return false;
    }
    // Fallback to legacy time_start/time_end/days
    if (!dDays(d).includes(wd)) return false;
    return isInTimeWindow(dTimeStart(d), dTimeEnd(d));
}

function isPublicHoliday() { try { return getState(CONFIG.holiday_dp).val === true; } catch (e) { return false; } }

function isHoliday() {
    try {
        const von = getState(CONFIG.holiday_from_dp).val, bis = getState(CONFIG.holiday_to_dp).val;
        if (!von || !bis) return false;
        const h = new Date().toISOString().slice(0, 10);
        return h >= von && h <= bis;
    } catch (e) { return false; }
}

function getDaysUntilHolidayEnd() {
    try {
        const bis = getState(CONFIG.holiday_to_dp).val;
        if (!bis) return -1;
        const end = new Date(bis + 'T00:00:00'), heute = new Date(); heute.setHours(0, 0, 0, 0);
        return Math.ceil((end - heute) / 86400000);
    } catch (e) { return -1; }
}

function isDeviceDisabledByHoliday(d) {
    if (!dHolidayDisable(d) || !isHoliday()) return false;
    const pre = dHolidayPre(d);
    if (pre > 0) { const left = getDaysUntilHolidayEnd(); if (left >= 0 && left <= pre) return false; }
    return true;
}

function isDeviceOn(d) {
    try {
        const v = getState(d.switch_dp).val;
        return v === true || v === 1 || v === '1' || v === 'true';
    }
    catch (e) { return false; }
}

// Verwende internen State wenn der externe DP nicht zuverlässig ist (z.B. Modbus/Tuya)
function isDeviceOnInternal(d) {
    const ds = state.device_states[d.id];
    return ds ? ds.is_on === true : isDeviceOn(d);
}

function isMinRuntimeMet(d) {
    const mr = dMinRuntime(d); if (!mr || mr <= 0) return true;
    const ds = state.device_states[d.id]; if (!ds || !ds.last_on) return true;
    return (Date.now() - ds.last_on) / 60000 >= mr;
}

function isMinPauseMet(d) {
    const mp = dMinPause(d); if (!mp || mp <= 0) return true;
    const ds = state.device_states[d.id]; if (!ds || !ds.last_off) return true;
    return (Date.now() - ds.last_off) / 60000 >= mp;
}

function switchOn(d, pw) {
    if (!isMinPauseMet(d)) return;
    const maxW = dPowerMax(d);
    if (d.is_dimmable && d.power_set_dp) setState(d.power_set_dp, Math.round(pw));
    // Modbus erwartet number, Shelly/Tuya boolean
    const onVal = d.switch_dp.startsWith('modbus.') ? 1 : true;
    setState(d.switch_dp, onVal);
    if (!state.device_states[d.id]) state.device_states[d.id] = {};
    state.device_states[d.id].is_on = true;
    state.device_states[d.id].last_on = Date.now();
    state.device_states[d.id].power_set = pw || maxW;
    const t = d.temp_sensor_dp ? ` | WW: ${getDeviceTemp(d).toFixed(1)}/${getDeviceTempTop(d).toFixed(1)}°C` : '';
    logInfo(`✓ ${dName(d)} EIN (${Math.round(pw || maxW)}W) | SOC: ${Math.round(getSOC())}% | PV: ${Math.round(getPVPower())}W${t}`);
}

function switchOff(d) {
    if (!isMinRuntimeMet(d)) return;
    if (d.is_dimmable && d.power_set_dp) setState(d.power_set_dp, 0);
    const offVal = d.switch_dp.startsWith('modbus.') ? 0 : false;
    setState(d.switch_dp, offVal);
    if (!state.device_states[d.id]) state.device_states[d.id] = {};
    state.device_states[d.id].is_on = false;
    state.device_states[d.id].last_off = Date.now();
    state.device_states[d.id].power_set = 0;
    const t2 = d.temp_sensor_dp ? ` | WW: ${getDeviceTemp(d).toFixed(1)}/${getDeviceTempTop(d).toFixed(1)}°C` : '';
    logInfo(`✗ ${dName(d)} AUS | SOC: ${Math.round(getSOC())}% | PV: ${Math.round(getPVPower())}W${t2}`);
}

function switchOffForce(d) {
    if (d.is_dimmable && d.power_set_dp) setState(d.power_set_dp, 0);
    const offVal = d.switch_dp.startsWith('modbus.') ? 0 : false;
    setState(d.switch_dp, offVal);
    if (!state.device_states[d.id]) state.device_states[d.id] = {};
    state.device_states[d.id].is_on = false;
    state.device_states[d.id].last_off = Date.now();
    state.device_states[d.id].power_set = 0;
    const t3 = d.temp_sensor_dp ? ` | WW: ${getDeviceTemp(d).toFixed(1)}/${getDeviceTempTop(d).toFixed(1)}°C` : '';
    logInfo(`⚠ ${dName(d)} SOFORT AUS | SOC: ${Math.round(getSOC())}% | PV: ${Math.round(getPVPower())}W${t3}`);
}

function adjustPower(d, pw) {
    if (!d.is_dimmable || !d.power_set_dp) return;
    const cl = Math.max(dPowerMin(d), Math.min(dPowerMax(d), Math.round(pw)));
    setState(d.power_set_dp, cl);
    if (!state.device_states[d.id]) state.device_states[d.id] = {};
    state.device_states[d.id].power_set = cl;
    logDebug(`${dName(d)}: ${cl}W`);
}

// Tatsächlichen Gerätestatus lesen und internen State synchronisieren
function syncActualState(d) {
    if (!state.device_states[d.id]) state.device_states[d.id] = {};
    const ds = state.device_states[d.id];

    // Switch-Status synchronisieren (für alle Geräte)
    try {
        const actualOn = isDeviceOn(d);
        if (ds.is_on !== actualOn) {
            logInfo(`${dName(d)}: Externe Status-Änderung ${ds.is_on ? 'EIN' : 'AUS'} → ${actualOn ? 'EIN' : 'AUS'}`);
            ds.is_on = actualOn;
            if (actualOn) { ds.last_on = Date.now(); ds.power_set = dPowerMax(d); }
            else { ds.last_off = Date.now(); ds.power_set = 0; }
        }
    } catch (e) { /* DP nicht erreichbar */ }

    // Leistung synchronisieren (nur dimmbare Geräte)
    if (d.is_dimmable && d.power_set_dp) {
        try {
            const actual = getState(d.power_set_dp).val || 0;
            if (Math.abs(actual - ds.power_set) > 10) {
                logInfo(`${dName(d)}: Externe Leistungs-Änderung ${ds.power_set}W → ${actual}W`);
                ds.power_set = actual;
            }
        } catch (e) { /* DP nicht erreichbar */ }
    }
}

// SILENT MODE & FORECAST

function handleSilentMode(d) {
    if (!d.silent_mode_dp || !isDeviceOn(d)) return;
    const rules = dSilentRules(d);
    if (!rules || rules.length === 0) return;
    const wd = getWeekday(); let silent = false;
    for (const r of rules) {
        if (r.days && r.days.includes(wd)) {
            silent = r.before ? (isInTimeWindow('00:00', r.before) ? r.silent : !r.silent) : r.silent;
            break;
        }
    }
    try {
        if (getState(d.silent_mode_dp).val !== silent) {
            setState(d.silent_mode_dp, silent);
            logDebug(`${dName(d)}: Silent ${silent ? 'EIN' : 'AUS'}`);
        }
        setState(`${DP_BASE}.${d.id}.silent_active`, silent, true);
    } catch (e) { }
}

function getPreheatingPower(d) {
    if (!dFcEnabled(d)) return 0;
    const s = dFcStart(d), e = dFcEnd(d);
    if (!s || !e || !isInTimeWindow(s, e)) return 0;
    const soc = getSOC();
    const minSoc = dFcMinSoc(d);
    if (soc < minSoc) return 0;
    const kwh = getForecastTodayKwh();
    const minW = dPowerMin(d), maxW = dPowerMax(d);

    // Mindest-Forecast prüfen (niedrigste Regel)
    const rules = dFcRules(d);
    if (!rules || rules.length === 0) return 0;
    const minForecast = Math.min(...rules.map(r => r.min_forecast_kwh));
    if (kwh < minForecast) return 0;

    // Dynamische Berechnung basierend auf Batterie-Ladeleistung
    // Idee: Was in die Batterie fließt können wir umleiten zum Heizstab
    const batCharge = getBatteryChargePower(); // positiv = Laden
    const surplus = Math.max(0, getSurplus());
    const PUFFER = 150; // W Puffer damit Batterie nicht ganz leer gesaugt wird

    // Verfügbar = was gerade in die Batterie fließt (minus Puffer) + ggf. Netz-Einspeisung
    const available = Math.max(0, batCharge - PUFFER) + surplus;

    // Bei gutem Forecast zusätzlich etwas Batterie-Entnahme erlauben
    const forecastFactor = Math.min(2.0, kwh / 20);
    const socReserve = Math.max(0, soc - minSoc);
    const batteryBonus = socReserve * 15 * forecastFactor; // reduziert (15 statt 30), da Ladeleistung schon genutzt wird

    const total = Math.round(Math.max(minW, Math.min(maxW, available + batteryBonus)));

    logDebug(`${dName(d)}: Vorheizen BatLade=${Math.round(batCharge)}W Surplus=${Math.round(surplus)}W Bonus=${Math.round(batteryBonus)}W → ${total}W (SOC ${Math.round(soc)}%, Forecast ${Math.round(kwh)}kWh)`);
    return total;
}

    for (const d of DEVICES) {
        const b = `${DP_BASE}.${d.id}`;

        // Identifikation (RW)
        createState(`${b}.name`, d.name, { name: 'Gerätename', type: 'string', role: 'text', def: d.name });
        createState(`${b}.mode`, 'auto', { name: 'Modus', type: 'string', role: 'text', states: { 'auto': 'Automatik', 'manual_on': 'Manuell EIN', 'manual_off': 'Manuell AUS', 'disabled': 'Deaktiviert' } });
        createState(`${b}.priority`, d.priority, { name: 'Priorität', type: 'number', role: 'value', min: 1, max: 99, def: d.priority });

        // Leistung (RW + R)
        createState(`${b}.power_min_w`, d.power_min_w, { name: 'Min. Leistung', type: 'number', unit: 'W', role: 'value.power', def: d.power_min_w });
        createState(`${b}.power_max_w`, d.power_max_w, { name: 'Max. Leistung', type: 'number', unit: 'W', role: 'value.power', def: d.power_max_w });
        createState(`${b}.is_dimmable`, d.is_dimmable, { name: 'Stufenlos regelbar', type: 'boolean', role: 'indicator', def: d.is_dimmable });

        // Überschuss-Schwellen (RW)
        createState(`${b}.min_soc`, d.min_soc, { name: 'Min. SOC Einschalten', type: 'number', unit: '%', def: d.min_soc, desc: 'SOC muss mindestens diesen Wert haben zum Einschalten' });
        createState(`${b}.soc_off`, d.soc_off, { name: 'SOC Ausschalten', type: 'number', unit: '%', def: d.soc_off, desc: 'Gerät wird ausgeschaltet wenn SOC unter diesen Wert fällt' });
        createState(`${b}.surplus_on_avg_w`, d.surplus_on_avg_w, { name: 'Überschuss Einschalten', type: 'number', unit: 'W', def: d.surplus_on_avg_w, desc: 'Ø Überschuss ab dem eingeschaltet wird' });
        createState(`${b}.surplus_off_avg_w`, d.surplus_off_avg_w, { name: 'Überschuss Ausschalten', type: 'number', unit: 'W', def: d.surplus_off_avg_w, desc: 'Ø Überschuss unter dem ausgeschaltet wird' });
        createState(`${b}.avg_minutes`, d.avg_minutes, { name: 'Durchschnitts-Zeitraum', type: 'number', unit: 'min', def: d.avg_minutes, desc: 'Minuten für Überschuss-Durchschnitt' });

        // Zeiten & Tage (RW)
        createState(`${b}.time_start`, d.time_start, { name: 'Zeitfenster Start', type: 'string', role: 'text', def: d.time_start, desc: 'Format HH:MM' });
        createState(`${b}.time_end`, d.time_end, { name: 'Zeitfenster Ende', type: 'string', role: 'text', def: d.time_end, desc: 'Format HH:MM' });
        createState(`${b}.days`, JSON.stringify(d.days), { name: 'Aktive Wochentage', type: 'string', role: 'json', def: JSON.stringify(d.days), desc: 'JSON: [1,2,3,4,5,6,7] (1=Mo, 7=So)' });
        createState(`${b}.exclude_holidays`, d.exclude_holidays, { name: 'An Feiertagen aus', type: 'boolean', role: 'switch', def: d.exclude_holidays });
        // Time rules (JSON: per-day scheduling)
        const defaultTimeRules = JSON.stringify([{ days: d.days, start: d.time_start, end: d.time_end }]);
        createState(`${b}.time_rules`, defaultTimeRules, { name: 'Zeitfenster-Regeln', type: 'string', role: 'json', def: defaultTimeRules, desc: 'JSON: [{"days":[1,2,3,4],"start":"10:00","end":"17:00"}]' });

        // Mindestlaufzeit (RW)
        createState(`${b}.min_runtime_min`, d.min_runtime_min, { name: 'Mindestlaufzeit', type: 'number', unit: 'min', def: d.min_runtime_min, desc: 'Min. Minuten nach Einschalten' });
        createState(`${b}.min_pause_min`, d.min_pause_min, { name: 'Mindestpause', type: 'number', unit: 'min', def: d.min_pause_min, desc: 'Min. Minuten nach Ausschalten' });

        // Benachrichtigungen (RW)
        createState(`${b}.notify`, d.notify, { name: 'Benachrichtigungen', type: 'boolean', role: 'switch', def: d.notify });
        createState(`${b}.max_notifications_per_day`, d.max_notifications_per_day, { name: 'Max. Benachrichtigungen/Tag', type: 'number', def: d.max_notifications_per_day });

        // Forecast (RW)
        createState(`${b}.forecast_enabled`, d.forecast_enabled, { name: 'Forecast-Vorheizen', type: 'boolean', role: 'switch', def: d.forecast_enabled });
        createState(`${b}.forecast_preheat_start`, d.forecast_preheat_start || '', { name: 'Vorheizen Start', type: 'string', def: d.forecast_preheat_start || '', desc: 'Format HH:MM' });
        createState(`${b}.forecast_preheat_end`, d.forecast_preheat_end || '', { name: 'Vorheizen Ende', type: 'string', def: d.forecast_preheat_end || '', desc: 'Format HH:MM' });
        createState(`${b}.forecast_preheat_min_soc`, d.forecast_preheat_min_soc, { name: 'Vorheizen Min. SOC', type: 'number', unit: '%', def: d.forecast_preheat_min_soc, desc: 'Vorheizen stoppt unter diesem SOC' });
        createState(`${b}.forecast_rules`, JSON.stringify(d.forecast_rules || []), { name: 'Forecast Regeln', type: 'string', role: 'json', def: JSON.stringify(d.forecast_rules || []), desc: 'JSON: [{"min_forecast_kwh":30,"power_w":1500}]' });

        // Urlaub (RW)
        createState(`${b}.disable_on_holiday`, d.disable_on_holiday, { name: 'Im Urlaub deaktivieren', type: 'boolean', role: 'switch', def: d.disable_on_holiday });
        createState(`${b}.holiday_pre_days`, d.holiday_pre_days, { name: 'Tage vor Urlaubsende', type: 'number', def: d.holiday_pre_days, desc: 'X Tage vor Ende wieder aktivieren' });

        // Silent Mode (RW + R)
        createState(`${b}.silent_rules`, JSON.stringify(d.silent_rules || []), { name: 'Silent Mode Regeln', type: 'string', role: 'json', def: JSON.stringify(d.silent_rules || []), desc: 'JSON: [{"days":[1,2,3,4],"before":"16:00","silent":true}]' });
        createState(`${b}.silent_active`, false, { name: 'Silent Mode aktiv', type: 'boolean', role: 'indicator' });

        // Temperatur-Abschaltung (RW)
        if (d.temp_sensor_dp) {
            // temp_max (unten) kommt vom Modbus (temp_max_dp), kein eigener DP nötig
            createState(`${b}.temp_max_top`, d.temp_max_top || 75, { name: 'Max. Temperatur oben', type: 'number', unit: '°C', def: d.temp_max_top || 75, desc: 'Sicherheitsgrenze oben - sofort abschalten' });
            createState(`${b}.temp_restart`, d.temp_restart || 45, { name: 'Wiedereinschalt-Temperatur', type: 'number', unit: '°C', def: d.temp_restart || 45, desc: 'Erst unter dieser Temperatur (unten) wieder einschalten' });
            createState(`${b}.temp_blocked`, false, { name: 'Temp-Sperre aktiv', type: 'boolean', role: 'indicator' });
        }

        // Status (R)
        createState(`${b}.is_on`, false, { name: 'Status', type: 'boolean', role: 'switch' });
        createState(`${b}.power_w`, 0, { name: 'Aktuelle Leistung', type: 'number', unit: 'W', role: 'value.power' });
        createState(`${b}.last_action`, '', { name: 'Letzte Aktion', type: 'string', role: 'text' });
        createState(`${b}.holiday_blocked`, false, { name: 'Urlaub blockiert', type: 'boolean', role: 'indicator' });
        createState(`${b}.notifications_today`, 0, { name: 'Benachrichtigungen heute', type: 'number', role: 'value' });
    }
    logInfo('Datenpunkte erstellt');
}

function measureSurplus() {
    const surplus = getSurplus();
    state.surplus_history.push({ ts: Date.now(), value: surplus });
    state.surplus_history = state.surplus_history.filter(v => v.ts >= Date.now() - 3600000);
    setState(`${DP_BASE}.surplus_current_w`, Math.round(surplus), true);
    setState(`${DP_BASE}.surplus_avg_w`, Math.round(getAvgSurplus(30)), true);
    setState(`${DP_BASE}.pv_power_w`, Math.round(getPVPower()), true);
    setState(`${DP_BASE}.battery_soc`, Math.round(getSOC()), true);
    setState(`${DP_BASE}.forecast_today_kwh`, Math.round(getForecastTodayKwh()), true);
    setState(`${DP_BASE}.is_holiday`, isHoliday(), true);
    setState(`${DP_BASE}.holiday_days_remaining`, getDaysUntilHolidayEnd(), true);
}

function runEnergyManager() {
    const soc = getSOC(), wd = getWeekday(), ph = isPublicHoliday(), cs = getSurplus();
    const sorted = [...DEVICES].sort((a, b) => dPriority(a) - dPriority(b));
    let avail = cs, totalP = 0; const active = [];

    for (const d of sorted) {
        const mode = dMode(d), on = isDeviceOn(d), avg = getAvgSurplus(dAvgMin(d));
        const nm = dName(d), minW = dPowerMin(d), maxW = dPowerMax(d), nfy = dNotify(d);
        // Tatsächlichen Gerätestatus synchronisieren (nicht bei disabled/offline)
        if (mode !== 'disabled') {
            const isOnline = !d.online_dp || (function() { try { return getState(d.online_dp).val; } catch(e) { return false; } })();
            if (isOnline) syncActualState(d);
        }

        // Urlaub
        const hb = isDeviceDisabledByHoliday(d);
        setState(`${DP_BASE}.${d.id}.holiday_blocked`, hb, true);
        if (hb && mode === 'auto') {
            if (on) {
                const ds = state.device_states[d.id];
                if (!ds || ds.is_on !== false) { switchOffForce(d); if (nfy) notify(d.id, `🏖️ ${nm} AUS (Urlaub)`); }
                else { const offV = d.switch_dp.startsWith('modbus.') ? 0 : false; setState(d.switch_dp, offV); }
            }
            continue;
        }

        // Temperatur-Abschaltung (höchste Priorität nach Urlaub)
        // Unten = Hauptsteuerung (wo der Heizstab sitzt) → Abschalten wenn unten >= Max
        // Oben = nur Notfall-Sicherheit bei >= 75°C (Verbrühschutz)
        // Wiedereinschalten wenn unten < Restart
        if (d.temp_sensor_dp) {
            const tempBottom = getDeviceTemp(d);
            const tempTop = getDeviceTempTop(d);
            const tMax = dTempMax(d);
            const tRestart = dTempRestart(d);
            const tMaxTop = dTempMaxTop(d);
            const tempBlocked = state.device_states[d.id] ? state.device_states[d.id].temp_blocked || false : false;

            if (tMax > 0 && (tempBottom >= tMax || tempTop >= tMaxTop) && !tempBlocked) {
                const reason = tempTop >= tMaxTop
                    ? `Sicherheit oben ${tempTop.toFixed(1)}°C >= ${tMaxTop}°C`
                    : `unten ${tempBottom.toFixed(1)}°C >= ${tMax}°C`;
                if (on) { switchOffForce(d); if (nfy) notify(d.id, `🌡️ ${nm} AUS (${reason})`); }
                if (state.device_states[d.id]) state.device_states[d.id].temp_blocked = true;
                setState(`${DP_BASE}.${d.id}.temp_blocked`, true, true);
                logInfo(`${nm}: Temp-Sperre aktiv (${reason}) [oben=${tempTop.toFixed(1)} unten=${tempBottom.toFixed(1)}]`);
                continue;
            }

            if (tempBlocked) {
                if (tRestart > 0 && tempBottom < tRestart) {
                    if (state.device_states[d.id]) state.device_states[d.id].temp_blocked = false;
                    setState(`${DP_BASE}.${d.id}.temp_blocked`, false, true);
                    logInfo(`${nm}: Temp-Sperre aufgehoben (unten ${tempBottom.toFixed(1)}°C < ${tRestart}°C) [oben=${tempTop.toFixed(1)}]`);
                } else {
                    // Noch gesperrt → nicht einschalten
                    if (on) switchOffForce(d);
                    setState(`${DP_BASE}.${d.id}.temp_blocked`, true, true);
                    continue;
                }
            }
        }

        // Manual
        if (mode === 'manual_on') {
            if (!on) switchOn(d, maxW);
            if (!state.device_states[d.id]) state.device_states[d.id] = {};
            state.device_states[d.id].is_on = true;
            state.device_states[d.id].power_set = maxW;
            totalP += maxW; active.push(`${nm} (manuell)`);
            handleSilentMode(d); continue;
        }
        if (mode === 'manual_off' || mode === 'disabled') {
            // Bei offline-Geräten (kein Strom) nicht versuchen zu schalten
            const isOnline = !d.online_dp || (function() { try { return getState(d.online_dp).val; } catch(e) { return false; } })();
            if (isOnline && on) {
                const ds = state.device_states[d.id];
                if (!ds || ds.is_on !== false) switchOffForce(d);
                else { const offV = d.switch_dp.startsWith('modbus.') ? 0 : false; setState(d.switch_dp, offV); }
            }
            continue;
        }

        // Auto-Checks
        if (!isInSchedule(d)) { if (on) { switchOffForce(d); if (nfy) notify(d.id, `${nm} AUS (Zeitfenster)`); } continue; }
        if (dExclHolidays(d) && ph) { if (on) switchOffForce(d); continue; }
        if (d.online_dp) { try { if (!getState(d.online_dp).val) continue; } catch (e) { continue; } }
        if (d.requires_dp) { try { if (getState(d.requires_dp).val !== d.requires_value) { if (on) { switchOffForce(d); if (nfy) notify(d.id, `${nm} AUS (Abhängigkeit)`); } continue; } } catch (e) { if (on) switchOffForce(d); continue; } }

        // Forecast
        const php = getPreheatingPower(d);
        if (php > 0) {
            if (!on) { switchOn(d, php); if (nfy) notify(d.id, `☀️ ${nm} VORHEIZEN ${php}W\nForecast: ${Math.round(getForecastTodayKwh())} kWh | SOC: ${Math.round(soc)}%`); totalP += php; active.push(`${nm} (Vorheizen ${php}W)`); }
            else if (d.is_dimmable) { const cp = (state.device_states[d.id] || {}).power_set || php; if (Math.abs(php - cp) > 50) adjustPower(d, php); totalP += php; active.push(`${nm} (Vorheizen ${php}W)`); }
            else { totalP += php; active.push(`${nm} (Vorheizen)`); }
            handleSilentMode(d); continue;
        }

        // Überschuss
        if (on) {
            if (d.is_dimmable) {
                const cp = (state.device_states[d.id] || {}).power_set || minW;
                const np = cp + avail;
                // FIX: Auch bei dimmbaren Geräten den Durchschnitt gegen surplus_off prüfen
                // UND SOC-Schwelle prüfen — vorher wurde nur np < minW geprüft
                if ((avg < dSurplusOff(d) || np < minW || soc < dSocOff(d)) && isMinRuntimeMet(d)) {
                    switchOff(d);
                    if (nfy) notify(d.id, `${nm} AUS\nØ ${Math.round(avg)}W | SOC: ${Math.round(soc)}%`);
                    avail += cp;
                }
                else if (np >= minW) { const cl = Math.max(minW, Math.min(maxW, np)); if (Math.abs(cl - cp) > 50) adjustPower(d, cl); avail -= (cl - cp); totalP += cl; active.push(`${nm} (${Math.round(cl)}W)`); }
                else { totalP += minW; active.push(`${nm} (${minW}W min.lauf)`); }
            } else {
                if ((avg < dSurplusOff(d) || soc < dSocOff(d)) && isMinRuntimeMet(d)) { switchOff(d); if (nfy) notify(d.id, `${nm} AUS\nØ ${Math.round(avg)}W | SOC: ${Math.round(soc)}%`); avail += maxW; }
                else { totalP += maxW; active.push(nm); avail -= maxW; }
            }
        } else {
            if (soc < dMinSoc(d)) continue;
            if (!isMinPauseMet(d)) continue;
            if (d.is_dimmable) {
                if (avg >= dSurplusOn(d)) { const sp = Math.min(maxW, Math.max(minW, avg)); switchOn(d, sp); if (nfy) notify(d.id, `${nm} EIN (${Math.round(sp)}W)\nØ ${Math.round(avg)}W | SOC: ${Math.round(soc)}%`); avail -= sp; totalP += sp; active.push(`${nm} (${Math.round(sp)}W)`); }
            } else {
                if (avg >= dSurplusOn(d)) { switchOn(d, maxW); if (nfy) notify(d.id, `${nm} EIN\nØ ${Math.round(avg)}W | SOC: ${Math.round(soc)}%`); avail -= maxW; totalP += maxW; active.push(nm); }
            }
        }
        handleSilentMode(d);
    }

    setState(`${DP_BASE}.managed_power_w`, Math.round(totalP), true);
    setState(`${DP_BASE}.active_devices`, active.join(', ') || 'keine', true);
    for (const d of DEVICES) {
        // Internen State bevorzugen (Modbus/Tuya melden nicht immer zuverlässig)
        const ds = state.device_states[d.id] || {};
        const on = ds.is_on === true ? true : isDeviceOn(d);
        const pw = on ? (ds.power_set || 0) : 0;
        setState(`${DP_BASE}.${d.id}.is_on`, on, true);
        setState(`${DP_BASE}.${d.id}.power_w`, pw, true);
        setState(`${DP_BASE}.${d.id}.last_action`, `${new Date().toLocaleString('de-DE')}: ${on ? 'EIN ' + pw + 'W' : 'AUS'}`, true);
    }
    logDebug(`Surplus: ${Math.round(cs)}W | Ø30: ${Math.round(getAvgSurplus(30))}W | Managed: ${Math.round(totalP)}W | ${active.join(', ') || 'keine'}`);
}

function setupDependencyWatchers() {
    for (const d of DEVICES) {
        if (d.requires_dp) {
            on({ id: d.requires_dp, change: 'ne' }, (obj) => {
                if (obj.state.val !== d.requires_value && isDeviceOn(d)) { switchOffForce(d); notify(d.id, `⚠️ ${dName(d)} SOFORT AUS!\nAbhängigkeit deaktiviert!`); }
            });
            logInfo(`Watcher: ${d.requires_dp} → ${dName(d)}`);
        }
    }
}

function midnightReset() {
    logInfo('=== Mitternacht-Reset ===');
    for (const d of DEVICES) {
        if (dMode(d) !== 'disabled') setState(`${DP_BASE}.${d.id}.mode`, 'auto');
        else logInfo(`${dName(d)}: bleibt 'disabled'`);
        if (isDeviceOn(d)) { switchOffForce(d); logInfo(`${dName(d)}: Reset → AUS`); }
    }
    state.surplus_history = [];
    logInfo('Reset abgeschlossen');
}

function startup() {
    logInfo('=============================================');
    logInfo('Energiemanager v1.7 gestartet');
    logInfo('=============================================');
    createDatapoints();
    for (const d of DEVICES) {
        // Switch-Datenpunkte beim Start auf korrekten Typ bereinigen
        try {
            const v = getState(d.switch_dp).val;
            if (d.switch_dp.startsWith('modbus.')) {
                // Modbus braucht number: true→1, false→0
                if (v === true) { setState(d.switch_dp, 1); logInfo(`${dName(d)}: Modbus Fix true→1`); }
                else if (v === false) { setState(d.switch_dp, 0); logInfo(`${dName(d)}: Modbus Fix false→0`); }
            }
        } catch (e) { /* DP nicht erreichbar */ }

        const devOn = isDeviceOn(d);
        // Wenn Gerät beim Start schon an ist, power_set aus dem Datenpunkt lesen
        let initPower = 0;
        if (devOn) {
            try { initPower = getState(`${DP_BASE}.${d.id}.power_w`).val || dPowerMax(d); }
            catch (e) { initPower = dPowerMax(d); }
        }
        // temp_blocked vom Datenpunkt lesen falls vorhanden
        let tempBlocked = false;
        if (d.temp_sensor_dp) {
            try { tempBlocked = getState(`${DP_BASE}.${d.id}.temp_blocked`).val === true; } catch (e) { }
        }
        state.device_states[d.id] = { is_on: devOn, last_on: devOn ? Date.now() : 0, last_off: 0, power_set: initPower, temp_blocked: tempBlocked };
        logInfo(`${dName(d)}: Prio ${dPriority(d)} | ${dPowerMin(d)}-${dPowerMax(d)}W | ${d.is_dimmable ? 'dimmbar' : 'on/off'} | ${state.device_states[d.id].is_on ? 'EIN' : 'AUS'}`);
    }
    setupDependencyWatchers();
    try { logInfo(`PV: ${Math.round(getPVPower())}W | Netz: ${Math.round(getSurplus())}W | SOC: ${Math.round(getSOC())}% | Forecast: ${Math.round(getForecastTodayKwh())} kWh | Urlaub: ${isHoliday() ? 'JA' : 'NEIN'}`); }
    catch (e) { logWarn(`Startup: ${e.message}`); }
    logInfo('=============================================');
    state.initialized = true;
}

startup();
schedule(`*/${CONFIG.messung_intervall_sek} * * * * *`, () => { measureSurplus(); });
schedule(`*/${CONFIG.pruef_intervall_sek} * * * * *`, () => { if (state.initialized) runEnergyManager(); });
schedule('0 0 * * *', () => { midnightReset(); });
